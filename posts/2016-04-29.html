(defun squre(x)
  (* x x))


(defun even?(x)
  (= (mod x 2) 0))


(defun doublef(x)
  (+ x x))


(defun half(x)
  (/ x 2))

(defun multiply-iterative(a b)
  (defun multiply-iter(product count temp)
    (progn (format t "~% ~d ~d ~d" product count temp )
    (cond ( (= count 0) temp)
	  ( (even?  count)   (multiply-iter (* 2 product)
					    (/ count 2)
					    temp))
	  ( (multiply-iter product
			   (- count 1)
			   (+ temp product)))  )))
  (multiply-iter a b 0))
